## Alguns comandos do GIT
os primeiros quando instala-lo é

# Para configurar o nome de quem editou os arquivos. pra saber que foi você sabe.
	git config --global user.name "Saulosantos17"
	git config --global user.email "saulosantos2002@gmail.com"

# Para iniciar o git no repositório sem o GitHub
	git init

# Mostra o que foi alterado no arquivo local para com o servidor
# O que tem de diferente no arquivo local do arquivo em nuvem
	git status

# Para adicionar as mudanças em todos os arquivos
	git add .

# Para adicionar as mudanças em um arquivoespecifico.txt
	git add arquivoespecifico.txt

# Para commitar as mudanças feita
	git commit -m "Nova Versão x.x.x"

# Para mandar as alterações para o servidor (GitHub)
	git push
